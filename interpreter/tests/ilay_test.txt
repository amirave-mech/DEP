function fibonacci(i)
    print(i)
    if (i = 0)
        print("is 0")
        return 0
    if (i = 1)
        print("is 1")
        return 1
    if (i = 2)
        print("is 2")
        return 1
    first <- fibonacci(i - 1)
    second <- fibonacci(i - 2)
    third <- first+second
    print(first)
    print(second)
    print(third)
    return third

print(fibonacci(3))
