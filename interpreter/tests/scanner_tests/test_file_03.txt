def greet(name):
    print("Hello,", name)

def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n - 1)

def main():
    var x = 5
    var y = 10
    if x < y:
        print("x is smaller than y")
    else:
        print("x is greater or equal to y")
    for i = 1; i <= 5; i = i + 1:
        print("Factorial of", i, "is", factorial(i))

greet("Alice")
main()
